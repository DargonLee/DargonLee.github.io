iOS-给UILabel增加复制粘贴功能

2019-06-11 iOS

第一步

//让label有资格成为第一响应者
- (BOOL)becomeFirstResponder
{
    return YES;
}

//label能执行哪些操作(比如 拷贝 等)
- (BOOL)canPerformAction:(SEL)action withSender:(id)sender
{
    if (action == @selector(copy:) || action == @selector(paste:) || action == @selector(cut:)) return YES;
    return NO;
}
